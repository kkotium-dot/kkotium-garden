model User {
  id        String    @id @default(cuid())
  email     String    @unique
  name      String?
  level     Int       @default(1)
  exp       Int       @default(0)
  createdAt DateTime  @default(now())
  updatedAt DateTime  @updatedAt
  products  Product[]
}

model Supplier {
  id           String    @id @default(cuid())
  name         String
  code         String    @unique
  abbr         String    @unique @db.VarChar(4)
  platformCode String    @default("ETC") @db.VarChar(10)
  platformUrl  String?
  contact      String?
  address      String?
  description  String?
  createdAt    DateTime  @default(now())
  updatedAt    DateTime  @updatedAt
  products     Product[]

  @@index([platformCode])
}

model OrderItem {
  id        String   @id @default(cuid())
  quantity  Int
  price     Int
  orderId   String
  productId String
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
  order     Order    @relation(fields: [orderId], references: [id])
  product   Product  @relation(fields: [productId], references: [id])

  @@index([orderId])
  @@index([productId])
}

model NaverCategory {

}

model OriginCode {
  id         String    @id @default(cuid())

  @@map("origin_codes")

model ShippingTemplate {
  id              String    @id @default(cuid())
  name            String
  code            String    @unique
  platformCode    String?   @map("platform_code")
  supplierCode    String?   @map("supplier_code")
  courierCode     String    @default("CJGLS") @map("courier_code")
  shippingType    Int       @default(1) @map("shipping_type")
  shippingFee     Int       @default(3000) @map("shipping_fee")
  freeThreshold   Int?      @map("free_threshold")
  returnFee       Int       @default(6000) @map("return_fee")
  exchangeFee     Int       @default(6000) @map("exchange_fee")
  jejuFee         Int       @default(5000) @map("jeju_fee")
  islandFee       Int       @default(5000) @map("island_fee")
  naverTemplateNo String?   @map("naver_template_no")
  bundleKey       String?   @map("bundle_key")

  createdAt       DateTime  @default(now()) @map("created_at")
  updatedAt       DateTime  @updatedAt @map("updated_at")
  Product         Product[]

  @@map("shipping_templates")
}

model daily_recommendations {
  id           String    @id @default(dbgenerated("gen_random_uuid()")) @db.Uuid
  date         DateTime  @db.Date
  product_name String    @db.VarChar(255)
  supplier_id  String?
  honey_score  Int
  season_tag   String?   @db.VarChar(50)
  status       String?   @default("pending") @db.VarChar(20)
  created_at   DateTime? @default(now()) @db.Timestamp(6)

  @@index([date], map: "idx_daily_recommendations_date")
  @@index([status], map: "idx_daily_recommendations_status")
}

model product_options {
  id            String    @id @default(dbgenerated("gen_random_uuid()")) @db.Uuid
  product_id    String    @unique
  option_type   String    @db.VarChar(50)
  option_names  Json?
  option_rows   Json?
  direct_inputs Json?
  created_at    DateTime? @default(now()) @db.Timestamp(6)
  updated_at    DateTime? @default(now()) @db.Timestamp(6)
  Product       Product   @relation(fields: [product_id], references: [id], onDelete: Cascade, onUpdate: NoAction, map: "fk_product_options_product")

  @@index([product_id], map: "idx_product_options_product_id")
  @@index([option_type], map: "idx_product_options_type")
}
